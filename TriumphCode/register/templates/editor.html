#include <stdio.h>
int main(){
    //correct code for problem 1..sum of digits.
    int test, num, sum;
    scanf("%d", &test);
    while(test--){
        sum=0;
       scanf("%d", &num);
       while(num){
           sum+=num%10;
           num/=10;
       }
       printf("%d\n", sum);
    }
    return 0;
}
